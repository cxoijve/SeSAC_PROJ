FUNCTION zfmss_get_storage_name.
*"----------------------------------------------------------------------
*"*"Local Interface:
*"  IMPORTING
*"     REFERENCE(IV_STOID) TYPE  ZTSSSTRG-STOID
*"  EXPORTING
*"     REFERENCE(EV_SNAME) TYPE  ZTSSSTRG-STONM
*"----------------------------------------------------------------------


  SELECT SINGLE stonm
    INTO ev_sname
    FROM ztssstrg
    WHERE stoid = iv_stoid.



ENDFUNCTION.
