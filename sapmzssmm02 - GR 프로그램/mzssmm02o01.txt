MODULE modify_screen_0100 OUTPUT.

  LOOP AT SCREEN.

    CASE screen-group1.
      WHEN 'GR1'.
        IF gv_screen = 1.
          screen-input = 1.
          screen-invisible = 0.
          screen-active = 1.
        ENDIF.
      WHEN 'GR2'.
        IF gv_screen = 1.
          screen-input = 0.
          screen-active = 0.
          screen-invisible = 1.
        ENDIF.
    ENDCASE.

    CASE screen-group2.
      WHEN 'RA1'.
        IF gv_radio_screen = 0.
          screen-input = 0.
          screen-invisible = 1.
        ENDIF.
      WHEN 'RA2'.
        IF gv_radio_screen = 1.
          screen-input = 0.
          screen-invisible = 1.
        ENDIF.
    ENDCASE.

*    IF ZSSSMM0210-VENID and gv_radio_sceen = 1.
*      screen-active = 0.
*    ENDIF.
    MODIFY SCREEN.
  ENDLOOP.

ENDMODULE.
