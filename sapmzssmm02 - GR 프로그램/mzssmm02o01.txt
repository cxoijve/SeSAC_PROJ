*                         CHANGING go_con_up go_alv_poitems,
*                         CHANGING go_con_down go_alv_confirm.

    WHEN OTHERS.
  ENDCASE.
ENDMODULE.
*&---------------------------------------------------------------------*
*&      Module  MODIFY_SCREEN_0100  OUTPUT
*&---------------------------------------------------------------------*
*       text
*----------------------------------------------------------------------*
MODULE modify_screen_0100 OUTPUT.

  LOOP AT SCREEN.

    CASE screen-group1.
      WHEN 'GR1'.
        IF gv_screen = 1.
          screen-input = 1.
          screen-invisible = 0.
          screen-active = 1.
        ENDIF.
      WHEN 'GR2'.
        IF gv_screen = 1.
          screen-input = 0.
          screen-active = 0.
          screen-invisible = 1.
        ENDIF.
    ENDCASE.

    CASE screen-group2.
      WHEN 'RA1'.
        IF gv_radio_screen = 0.
          screen-input = 0.
          screen-invisible = 1.
        ENDIF.
      WHEN 'RA2'.
        IF gv_radio_screen = 1.
          screen-input = 0.
          screen-invisible = 1.
        ENDIF.
    ENDCASE.

    MODIFY SCREEN.
  ENDLOOP.

ENDMODULE.

MODULE modify_screen_0200 OUTPUT.
  LOOP AT SCREEN.
    CASE screen-group1.
      WHEN 'GR1'.
        IF gv_bbtn = 1.
          screen-input = 1.
          screen-invisible = 0.
          screen-active = 1.

        ELSEIF gv_bbtn = 0.

          screen-invisible = 1.

    ENDCASE.

    MODIFY SCREEN.
  ENDLOOP.
