FUNCTION Z_FMSS_GET_MAT_NAME.
*"--------------------------------------------------------------------
*"*"Local Interface:
*"  IMPORTING
*"     REFERENCE(IV_MATID) TYPE  ZESSMATNR
*"  EXPORTING
*"     REFERENCE(EV_MNAME) TYPE  ZESSMATNM
*"--------------------------------------------------------------------

  SELECT SINGLE matnm
      INTO ev_mname
      FROM ztssmat_t
      WHERE matnr = iv_matid.
  IF sy-subrc <> 0.
    RETURN.
  ENDIF.



ENDFUNCTION.
