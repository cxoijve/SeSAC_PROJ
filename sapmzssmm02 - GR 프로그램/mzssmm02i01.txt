MODULE user_command_0100 INPUT.

  CASE ok_code.
    WHEN 'BACK'.
      LEAVE TO SCREEN 0.
    WHEN 'SEARCH'.
    WHEN 'TAB1' OR 'TAB2'.
      IF ok_code = 'TAB1'. gv_screen = 1. ELSE. gv_screen = 0. ENDIF.
      t_info-activetab = ok_code.

    WHEN 'RA1'.
      IF gv_radio_vname = 'X'. gv_radio_screen  = 0. ELSEIF gv_radio_venid = 'X'. gv_radio_screen  = 1. ENDIF.
    WHEN 'P_BTN'.
      gv_entire_screen = 1.
    WHEN 'B_BTN'.
      gv_entire_screen = 0.
    WHEN OTHERS.
  ENDCASE.

ENDMODULE.
*&---------------------------------------------------------------------*
*&      Module  EXIT_0100  INPUT
*&---------------------------------------------------------------------*
*       text
*----------------------------------------------------------------------*
MODULE exit_0100 INPUT.

  CASE OK_CODE.
    WHEN 'EXIT'.
      leave PROGRAM.
    WHEN 'CANC'.
      leave to SCREEN 0.
    WHEN OTHERS.
  ENDCASE.

ENDMODULE.
