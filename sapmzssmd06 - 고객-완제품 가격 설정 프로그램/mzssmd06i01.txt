MODULE user_command_0100 INPUT.

  CASE ok_code.
    WHEN 'BACK'.
      LEAVE TO SCREEN 0.


    WHEN 'SELL'.
      PERFORM get_all_table_data.
      PERFORM get_all_prdnr_data.
      CALL SCREEN 101 STARTING AT 85 3 ENDING AT 120 26.


    WHEN 'CHANGE'.
      PERFORM get_detail_info_100 CHANGING gv_subrc.

      PERFORM set_update_field.
      IF gv_subrc IS INITIAL.
        PERFORM check_so_somnr_data USING gs_data-cstid
                                         gs_data-prdnr
                                CHANGING gv_subrc.
        IF gv_subrc <> 0.
          MESSAGE i401(zmcss).
          RETURN.
        ENDIF.

        IF gs_data-del = 'X'.
          MESSAGE i999(zmcss) WITH '현재 판매중인 업체만 선택해 주세요'.
          RETURN.
        ENDIF.
        CALL SCREEN 300.
        PERFORM check_alv_condition.
        PERFORM get_prdnr_price_data.
      ENDIF.


    WHEN 'SEARCH'.
      PERFORM check_alv_condition.
      PERFORM get_prdnr_price_data.


    WHEN 'RELEASE'.
      PERFORM get_detail_info_100 CHANGING gv_subrc.

      IF gv_subrc IS INITIAL.
        PERFORM check_so_somnr_data USING gs_data-cstid
                                          gs_data-prdnr
                                 CHANGING gv_subrc.
        IF gv_subrc <> 0.
          MESSAGE i401(zmcss).
          RETURN.
        ENDIF.


        IF gs_data-del = 'X'.
          MESSAGE i999(zmcss) WITH '활성화된 데이터를 선택해주세요'.
          RETURN.
        ENDIF.


        CALL SCREEN 301 STARTING AT 30 2 ENDING AT 80 5.
        PERFORM check_alv_condition.
        PERFORM get_prdnr_price_data.
      ENDIF.



    WHEN 'RESTORE'.
      PERFORM get_detail_info_100 CHANGING gv_subrc.
      IF gv_subrc IS INITIAL.
        IF gs_data-del <> 'X'.
          MESSAGE i999(zmcss) WITH '삭제된 데이터를 선택해주세요'.
          RETURN.
        ENDIF.
        CALL SCREEN 302 STARTING AT 30 2 ENDING AT 80 5.
        PERFORM check_alv_condition.
        PERFORM get_prdnr_price_data.
      ENDIF.

    WHEN 'ENTER'.
*	WHEN OTHERS.
  ENDCASE.

ENDMODULE.

MODULE user_command_0101 INPUT.

  CASE ok_code.
    WHEN 'CHECK'.
      PERFORM get_detail_info CHANGING gv_subrc.
      IF gv_subrc IS INITIAL.
        zsssmd06005-prdnr = gs_modal-prdnr.
        zsssmd06005-curky = 'KRW'.
        PERFORM get_matched_customer_data USING zsssmd06005-prdnr.
        CALL SCREEN 200.
        PERFORM check_alv_condition.
        PERFORM get_prdnr_price_data.
        CLEAR : gt_prdnr,
                gs_prdnr.
        LEAVE TO SCREEN 0.
      ENDIF.

    WHEN 'CANC'.
      LEAVE TO SCREEN 0.

    WHEN 'ENTER'.
  ENDCASE.
ENDMODULE.
