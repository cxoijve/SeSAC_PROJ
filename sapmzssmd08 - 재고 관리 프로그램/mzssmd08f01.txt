*&---------------------------------------------------------------------*
*&      Form  GET_INFO_0100
*&---------------------------------------------------------------------*
*       text
*----------------------------------------------------------------------*
*      -->P_GT_INFO_0100  text
*      -->P_ZSSSMD0801  text
*----------------------------------------------------------------------*
FORM get_info_0100  TABLES   pt_info_0100 STRUCTURE gs_info_0100
                    USING    VALUE(ps_cond_0100) TYPE zsssmd0801
                    CHANGING ps_alvtitle_0100 TYPE zsssmd0802
                             ps_sideinfo_0100 TYPE zsssmd0804
                             pv_subrc LIKE sy-subrc.

  CLEAR : pv_subrc, ps_alvtitle_0100, ps_sideinfo_0100, pt_info_0100[].

  " alv title 정보 획득
  MOVE-CORRESPONDING ps_cond_0100 TO ps_alvtitle_0100.

  CONCATENATE
    '플랜트'
    ps_alvtitle_0100-plant
    '창고'
    ps_alvtitle_0100-stoid
    '의 재고 현황'
    INTO ps_alvtitle_0100-infotext
    SEPARATED BY space.

  " side info 획득
  SELECT SINGLE a~plant a~pname
    a~ppern AS plnr
    b~lname AS pllnm
    b~fname AS plfnm
    b~telnr AS pltel
    b~persg AS plsg
    FROM ztssplant AS a
    INNER JOIN ztssemp AS b
    ON a~ppern EQ b~pernr
    INTO CORRESPONDING FIELDS OF ps_sideinfo_0100
    WHERE plant EQ ps_cond_0100-plant.

  IF sy-subrc <> 0.
    pv_subrc = 4.
    RETURN.
  ENDIF.

  " side info 획득
  SELECT SINGLE a~stoid a~stonm
    a~stadm AS stnr
    b~lname AS stlnm
    b~fname AS stfnm
    b~telnr AS sgtel
    b~persg AS stsg
    FROM ztssstrg AS a
    INNER JOIN ztssemp AS b
    ON a~stadm EQ b~pernr
    INTO CORRESPONDING FIELDS OF ps_sideinfo_0100
    WHERE stoid EQ ps_cond_0100-stoid.

  IF sy-subrc <> 0.
    pv_subrc = 4.
    RETURN.
  ENDIF.

  "DOMFIX 가져오기
  PERFORM get_domain_fixed_value TABLES gt_dd07v
                                 USING  ps_sideinfo_0100-plsg
                                        'ZDSSPERSG'
                                 CHANGING ps_sideinfo_0100-plsg_t.

  PERFORM get_domain_fixed_value TABLES gt_dd07v
                                 USING  ps_sideinfo_0100-stsg
                                        'ZDSSPERSG'
                                 CHANGING ps_sideinfo_0100-stsg_t.

  " 사원 이름 CONCAT
  CONCATENATE ps_sideinfo_0100-pllnm ps_sideinfo_0100-plfnm
  INTO ps_sideinfo_0100-planm.

  CONCATENATE ps_sideinfo_0100-stlnm ps_sideinfo_0100-stfnm
  INTO ps_sideinfo_0100-stanm.

  " pt_info_0100 획득

  IF ps_cond_0100-matnm IS NOT INITIAL.
    SELECT a~matnr b~matnm a~meins
      FROM ztssware AS a
      INNER JOIN ztssmat_t AS b
      ON a~matnr = b~matnr
      AND b~langu = sy-langu
      INTO CORRESPONDING FIELDS OF TABLE pt_info_0100
      WHERE stoid = ps_cond_0100-stoid
      AND matnm LIKE ps_cond_0100-matnm.
  ELSE.
    SELECT a~matnr b~matnm a~meins
      FROM ztssware AS a
      INNER JOIN ztssmat_t AS b
      ON a~matnr = b~matnr
      AND b~langu = sy-langu
      INTO CORRESPONDING FIELDS OF TABLE pt_info_0100
      WHERE stoid = ps_cond_0100-stoid.
  ENDIF.

  IF pt_info_0100[] IS INITIAL.
    pv_subrc = 4.
    RETURN.
  ENDIF.
  SORT pt_info_0100 BY matnr DESCENDING.
  DELETE ADJACENT DUPLICATES FROM pt_info_0100 COMPARING matnr.


  DATA : lt_color TYPE lvc_t_scol WITH HEADER LINE.

  LOOP AT pt_info_0100.

    CALL FUNCTION 'Z_FMSS_GET_AVAIL_STOCKS'
      EXPORTING
        iv_begda         = '11110101'
        iv_endda         = '99991231'
        iv_stoid         = ps_cond_0100-stoid
        iv_matnr         = pt_info_0100-matnr
      IMPORTING
        ev_astck         = pt_info_0100-astck
        ev_istck         = pt_info_0100-istck
        ev_ostck         = pt_info_0100-ostck
        ev_dstck         = pt_info_0100-dstck
        ev_fstck         = pt_info_0100-fstck
      EXCEPTIONS
        material_unfound = 1
        minus_avail      = 2
        query_failed     = 3
        OTHERS           = 4.
    IF sy-subrc <> 0.
      pv_subrc = 4.
      RETURN.
    ENDIF.

    IF pt_info_0100-astck IS NOT INITIAL.
      lt_color-fname = 'ASTCK'.
      lt_color-color-int = 1.
      lt_color-color-inv = 0.
      lt_color-color-col = 5.
      APPEND lt_color.
    ENDIF.

    IF pt_info_0100-fstck IS NOT INITIAL.
      lt_color-fname = 'FSTCK'.
      lt_color-color-int = 0.
      lt_color-color-inv = 0.
      lt_color-color-col = 6.
      APPEND lt_color.

    ENDIF.

    APPEND LINES OF lt_color TO pt_info_0100-cellscol.
    MODIFY pt_info_0100 TRANSPORTING astck istck ostck dstck fstck cellscol.
    CLEAR : lt_color, lt_color[].
  ENDLOOP.

ENDFORM.

*&---------------------------------------------------------------------*
*&      Form  CALC_DATUM
*&---------------------------------------------------------------------*
*       text
*----------------------------------------------------------------------*
*  -->  p1        text
*  <--  p2        text
*----------------------------------------------------------------------*
FORM get_calc_datum  USING    VALUE(p_datum)
                              VALUE(p_signum)
                              VALUE(p_years)
                              VALUE(p_months)
                              VALUE(p_days)
                    CHANGING p_begda LIKE p0001-begda.

  DATA: lv_date   TYPE  p0001-begda,
        lv_days   TYPE  t5a4a-dlydy,
        lv_months TYPE  t5a4a-dlymo,
        lv_signum TYPE  t5a4a-split,
        lv_years  TYPE  t5a4a-dlyyr.

  IF p_signum IS INITIAL.
    lv_signum = '+'.
  ELSE.
    lv_signum = p_signum.
  ENDIF.
  lv_years = p_years.
  lv_months = p_months.
  lv_days = p_days.

  lv_date = p_datum.
  CALL FUNCTION 'RP_CALC_DATE_IN_INTERVAL'
    EXPORTING
      date      = lv_date
      days      = lv_days
      months    = lv_months
      signum    = lv_signum
      years     = lv_years
    IMPORTING
      calc_date = p_begda.

ENDFORM.

*&---------------------------------------------------------------------*
*&      Form  SELECT_ROW_0100
*&---------------------------------------------------------------------*
*       text
*----------------------------------------------------------------------*
*      -->P_GT_INFO_0100  text
*      <--P_LS_ROW_0100  text
*      <--P_GV_SUBRC  text
*----------------------------------------------------------------------*
FORM select_row_0100  TABLES   pt_info_0100 STRUCTURE gs_info_0100
                      CHANGING ps_row_0100 LIKE gs_info_0100
                               pv_subrc LIKE sy-subrc.

  CLEAR : pv_subrc, ps_row_0100.

  DATA : lt_row   TYPE lvc_t_roid,
         ls_row   LIKE LINE OF lt_row,
         lv_index TYPE i.

  CALL METHOD go_alv_0100->get_selected_rows
    IMPORTING
      et_row_no = lt_row.

  DESCRIBE TABLE lt_row LINES lv_index.

  CASE lv_index.
    WHEN 1.
      pv_subrc = 0.
    WHEN 0.
      MESSAGE i999(zmcss) WITH '상세 조회를 위해 하나만 선택해 주세요.'.
      pv_subrc = 4.
      RETURN.
    WHEN OTHERS.
      MESSAGE i999(zmcss) WITH '상세 조회를 위해 하나만 선택해 주세요.'.
      pv_subrc = 4.
      RETURN.
  ENDCASE.

  READ TABLE lt_row INDEX 1 INTO ls_row.
  READ TABLE pt_info_0100 INDEX ls_row-row_id.

  IF sy-subrc <> 0.
    MESSAGE i999(zmcss) WITH '치명적인 에러가 발생했습니다. 선택한 Row를 Internal Table에서 찾을 수 없습니다.'.
    pv_subrc = 4.
    RETURN.
  ENDIF.

  MOVE-CORRESPONDING pt_info_0100 TO ps_row_0100.
  pv_subrc = 0.
ENDFORM.

        lt_color-color-col = 4.

        lt_color-fname = 'AMOUNT'.
        lt_color-color-int = 1.
        lt_color-color-inv = 0.
        lt_color-color-col = 4.
        APPEND lt_color.

        lt_color-fname = 'AMOUNT'.
        lt_color-color-int = 0.
        lt_color-color-inv = 0.
        lt_color-color-col = 4.
        APPEND lt_color.

    ENDCASE.
    APPEND LINES OF lt_color TO pt_info_0210-cellscol.
    CLEAR : lt_color[].

    APPEND LINES OF lt_color TO pt_info_0220-cellscol.

*&      Form  GET_INFO_0300
*&---------------------------------------------------------------------*
*       text
*----------------------------------------------------------------------*
*      -->P_GT_INFO_0300  text
*      -->P_ZSSSMD0810  text
*      <--P_GV_SUBRC  text
*----------------------------------------------------------------------*
FORM get_info_0300  TABLES   pt_info_0300 STRUCTURE gs_info_0300
                    USING    VALUE(ps_cond_0300) TYPE zsssmd0810
                    CHANGING pv_subrc LIKE gv_subrc.

  CLEAR : pv_subrc, pt_info_0300.

  DATA : lr_status TYPE RANGE OF tbtco-status,
         lr_datum  TYPE RANGE OF tbtco-sdlstrtdt.

  DATA : lv_subrc LIKE sy-subrc.

  PERFORM check_cond_0300 USING ps_cond_0300
                          CHANGING lv_subrc.

  IF lv_subrc <> 0. pv_subrc = 4. ENDIF.

  PERFORM set_range_0300 TABLES lr_status
                                lr_datum
                         USING  ps_cond_0300
                         CHANGING lv_subrc.

  IF lv_subrc <> 0. MESSAGE i999(zmcss) WITH '조회 조건 설정 중 에러가 발생했습니다.'. pv_subrc = 4. ENDIF.

  SELECT jobname sdlstrtdt sdlstrttm status
  FROM tbtco
  INTO CORRESPONDING FIELDS OF TABLE pt_info_0300
  WHERE sdluname EQ sy-uname
    AND sdlstrtdt IN lr_datum
    AND status IN lr_status.

  IF sy-subrc <> 0. MESSAGE s999(zmcss) WITH '조회 조건을 만족하는 데이터가 없습니다.'. pv_subrc = 4. ENDIF.

  PERFORM modify_info_0300 TABLES pt_info_0300
                           CHANGING lv_subrc.

  IF lv_subrc <> 0. MESSAGE i999(zmcss) WITH '데이터를 불러오는 도중 에러가 발생했습니다.'. pv_subrc = 4. ENDIF.

ENDFORM.

*&---------------------------------------------------------------------*
*       text
*----------------------------------------------------------------------*
*      -->P_PS_COND_0300  text
*      <--P_LV_SUBRC  text
*----------------------------------------------------------------------*
FORM check_cond_0300  USING    VALUE(ps_cond_0300) TYPE zsssmd0810
                      CHANGING pv_subrc LIKE sy-subrc.

  CLEAR pv_subrc.

  IF ps_cond_0300-begda IS INITIAL
    OR ps_cond_0300-endda IS INITIAL.
    MESSAGE i999(zmcss) WITH '날짜를 입력해 주십시오'.
    pv_subrc = 4.
    RETURN.
  ENDIF.

  IF ps_cond_0300-begda > ps_cond_0300-endda.
    MESSAGE i999(zmcss) WITH '시작일은 종료일보다 작거나 같아야 합니다'.
    pv_subrc = 4.
    RETURN.
  ENDIF.

  IF ps_cond_0300-isa IS INITIAL
    AND ps_cond_0300-isr IS INITIAL
    AND ps_cond_0300-isf IS INITIAL.

    MESSAGE i999(zmcss) WITH '하나 이상의 생산 계획 유형을 선택해 주십시오.'.
    pv_subrc = 4.
    RETURN.
  ENDIF.
ENDFORM.

*      -->P_LR_DATUM  text
*      -->P_PS_COND_0300  text

  pr_datum-option = 'BT'.
  pr_datum-low = ps_cond_0300-begda.

    pr_status-option = 'EQ'.
    pr_status-low = 'S'.
    APPEND pr_status.

    CLEAR pr_status.
    pr_status-sign = 'I'.
    pr_status-option = 'EQ'.
    pr_status-low = 'F'.
    APPEND pr_status.
  ENDIF.

  pv_subrc = 0.

*&---------------------------------------------------------------------*
*&      Form  MODIFY_SCREEN_0310
*&---------------------------------------------------------------------*
*       text
*----------------------------------------------------------------------*
*  -->  p1        text
*  <--  p2        text
*----------------------------------------------------------------------*
FORM modify_screen_0310 .
  LOOP AT SCREEN.
    CASE gv_fix_0310.
      WHEN 'X'. " 아이템 결정 체크함
        CASE screen-group1.
          WHEN 'PRD'.
            screen-input = '0'.
          WHEN 'FIX'.
            screen-active = '0'.
          WHEN 'CHG'.
            screen-active = '1'.
        ENDCASE.
      WHEN ''. " 아이템 결정 체크 안함.
        CASE screen-group1.
          WHEN 'PRD'.
            screen-input = '1'.
          WHEN 'FIX'.
            screen-active = '1'.
          WHEN 'CHG'.
            screen-active = '0'.
        ENDCASE.
      WHEN OTHERS.
        MESSAGE a999(zmcss) WITH 'UNKNOWN FIXFLAG!'.
    ENDCASE.

    CASE gv_imm_0310.
      WHEN 'X'. " 즉시 생산 체크함
        CASE screen-group1.
          WHEN 'PLA'.
            screen-active = '0'.
        ENDCASE.
      WHEN ''. " 즉시 생산 체크 안함.
        CASE screen-group1.
          WHEN 'PLA'.
            screen-active = '1'.
        ENDCASE.
      WHEN OTHERS.
        MESSAGE a999(zmcss) WITH 'UNKNOWN FIXFLAG!'.
    ENDCASE.


    MODIFY SCREEN.
  ENDLOOP.
ENDFORM.

*&---------------------------------------------------------------------*
*&      Form  GET_INFO_0310
*&---------------------------------------------------------------------*
*       text
*----------------------------------------------------------------------*
*      -->P_GT_INFO_0310  text
*      -->P_ZSSSMD0813  text
*      <--P_ZSSSMD0813_MEINS  text
*      <--P_GV_SUBRC  text
*----------------------------------------------------------------------*
FORM get_info_0310  TABLES   pt_info_0310 STRUCTURE gs_info_0310
                    USING    VALUE(ps_cond_0310) TYPE zsssmd0813
                    CHANGING pv_meins TYPE zessmeins
                             pv_infotext TYPE ddtext
                             pv_subrc LIKE sy-subrc.

  CLEAR : pv_subrc , pv_meins.

  pv_meins = 'EA'.

  DATA lv_subrc LIKE sy-subrc.
  DATA lt_req TYPE TABLE OF ztssreq WITH HEADER LINE.
  DATA : lt_color TYPE lvc_t_scol WITH HEADER LINE.

  " INFO TEXT 용
  DATA : lv_befamt   TYPE int4,
         lv_aftamt   TYPE int4,
         lv_befamt_t TYPE string,
         lv_aftamt_t TYPE string.

  PERFORM check_cond_0310 USING ps_cond_0310
                          CHANGING lv_subrc.

  IF lv_subrc <> 0. pv_subrc = 4. RETURN. ENDIF.

  " SET INFOTEXT
  CALL FUNCTION 'Z_FMSS_GET_AVAIL_STOCKS'
    EXPORTING
      iv_begda         = '11110101'
      iv_endda         = '99991231'
      iv_stoid         = 'SL02'
      iv_matnr         = ps_cond_0310-matnr
    IMPORTING
      ev_astck         = lv_befamt
    EXCEPTIONS
      material_unfound = 1
      minus_avail      = 2
      query_failed     = 3
      OTHERS           = 4.
  IF sy-subrc <> 0. pv_subrc = 4. RETURN. ENDIF.

  lv_aftamt = lv_befamt + ps_cond_0310-amount.
  lv_befamt_t = lv_befamt.
  lv_aftamt_t = lv_aftamt.

  CONCATENATE
    lv_befamt_t
    'EA'
    '→'
    lv_aftamt_t
    'EA'
  INTO pv_infotext
  SEPARATED BY space.

  SELECT *
    FROM ztssreq
    INTO CORRESPONDING FIELDS OF TABLE lt_req
    WHERE prdnr EQ ps_cond_0310-matnr.

  IF sy-subrc <> 0. MESSAGE i999(zmcss) WITH '자재' ps_cond_0310-matnr '의 BOM이 존재하지 않습니다.'. pv_subrc = 4. RETURN. ENDIF.

  LOOP AT lt_req.
    pt_info_0310-matnr = lt_req-rawnr.
    pt_info_0310-ustck = lt_req-amount * ps_cond_0310-amount.
    pt_info_0310-meins = lt_req-meins.

    " Need enhancement
    pt_info_0310-stoid = 'SL01'.
    pt_info_0310-stonm = '제 1 원자재 창고'.

    SELECT SINGLE matnm
      FROM ztssmat_t
      INTO pt_info_0310-matnm
      WHERE matnr EQ pt_info_0310-matnr
      AND langu EQ sy-langu.

    CALL FUNCTION 'Z_FMSS_GET_AVAIL_STOCKS'
      EXPORTING
        iv_begda         = '11110101'
        iv_endda         = '99991231'
        iv_stoid         = 'SL01'
        iv_matnr         = pt_info_0310-matnr
      IMPORTING
        ev_astck         = pt_info_0310-astck
      EXCEPTIONS
        material_unfound = 1
        minus_avail      = 2
        query_failed     = 3
        OTHERS           = 4.
    IF sy-subrc <> 0. pv_subrc = 4. RETURN. ENDIF.

    pt_info_0310-afamt = pt_info_0310-astck - pt_info_0310-ustck.

    IF pt_info_0310-afamt < 0.
      lt_color-fname = 'AFAMT'.
      lt_color-color-int = 1.
      lt_color-color-inv = 0.
      lt_color-color-col = 6.
      APPEND lt_color.
      MESSAGE s999(zmcss) WITH '원자재 재고가 부족합니다.'.
    ELSE.
      lt_color-fname = 'AFAMT'.
      lt_color-color-int = 1.
      lt_color-color-inv = 0.
      lt_color-color-col = 5.
      APPEND lt_color.
    ENDIF.

    APPEND LINES OF lt_color TO pt_info_0310-cellscol.
    APPEND pt_info_0310.
    CLEAR : pt_info_0310-cellscol[], pt_info_0310, lt_color[].
  ENDLOOP.

  pv_subrc = 0.

ENDFORM.

*----------------------------------------------------------------------*

*      -->P_ZSSSMD0812  text
*      -->P_ZSSSMD0813  text
*      <--P_GS_INWARE  text
*      <--P_GV_SUBRC  text
*----------------------------------------------------------------------*
FORM check_mrp  TABLES   pt_info_0310 STRUCTURE gs_info_0310
                         pt_outwares STRUCTURE ztssware
                USING    VALUE(ps_cond_time) TYPE zsssmd0812
                         VALUE(ps_cond_prd) TYPE zsssmd0813
                CHANGING ps_inware TYPE ztssware
                         pv_subrc LIKE sy-subrc.

    MESSAGE i999(zmcss) WITH '완제품을 결정해 주십시오.'.

    RETURN.
  ENDIF.

  LOOP AT pt_info_0310.
    IF pt_info_0310-afamt < 0.
      MESSAGE i999(zmcss) WITH '자재' pt_info_0310-matnm '의 수량이 부족합니다.'.
      pv_subrc = 4.
      RETURN.
    ENDIF.
  ENDLOOP.

  IF ps_cond_time-isbook = 'X'
    AND ( ps_cond_time-sdlstrtdt IS INITIAL
    OR ps_cond_time-sdlstrttm IS INITIAL ).

    MESSAGE i999(zmcss) WITH '자재 생산 계획의 일자 및 시간을 입력해 주십시오.'.
    pv_subrc = 4.
    RETURN.
  ENDIF.

  IF ps_cond_time-isbook = 'X'
    AND ( ps_cond_time-sdlstrtdt < sy-datum
    OR ( ps_cond_time-sdlstrtdt EQ sy-datum
    AND ps_cond_time-sdlstrttm < sy-uzeit ) ).

    MESSAGE i999(zmcss) WITH '과거 시점으로의 생산 계획은 생성할 수 없습니다'.
    pv_subrc = 4.
    RETURN.
  ENDIF.

*  ps_cond_prd-amount
*  ps_cond_prd-matnr
*  ps_cond_prd-meins
*  ps_cond_prd-stoid

  " INWARE SETTINGS
  MOVE-CORRESPONDING ps_cond_prd TO ps_inware.
  ps_inware-wtype = 'I'.

  CASE ps_cond_time-isbook.
    WHEN 'X'.
      ps_inware-wdate = ps_cond_time-sdlstrtdt.
    WHEN OTHERS.
      ps_inware-wdate = sy-datum.
  ENDCASE.

  ps_inware-rgtor = 'E0000024'.

  " OUTWARE SETTINGS
  LOOP AT pt_info_0310.

*     pt_info_0310-ustck
*     pt_info_0310-matnr
*     pt_info_0310-meins
*     pt_info_0310-stoid

    pt_outwares-wtype = 'O'.

      WHEN 'X'.
        pt_outwares-wdate = ps_cond_time-sdlstrtdt.
      WHEN OTHERS.
        pt_outwares-wdate = sy-datum.
    ENDCASE.

    pt_outwares-rgtor = 'E0000012'.
    APPEND pt_outwares.
  ENDLOOP.

  pv_subrc = 0.
ENDFORM.

*      <--P_GV_SUBRC  text
*----------------------------------------------------------------------*
FORM get_info_0400_from_0220  TABLES   pt_info_0400 STRUCTURE gs_info_0400
                                       pt_info_0220 STRUCTURE gs_info_0200
                              CHANGING ps_row_0100 LIKE gs_info_0100
                                       pv_subrc LIKE sy-subrc.

  CLEAR : pv_subrc, pt_info_0400[], ps_row_0100.
  READ TABLE pt_info_0220 INDEX 1.

  IF sy-subrc <> 0.
    MESSAGE i999(zmcss) WITH '선택된 행이 없습니다'.
    pv_subrc = 4.
    RETURN.
  ENDIF.

  SELECT SINGLE a~matnr b~matnm
    FROM ztssware AS a
    INNER JOIN ztssmat_t AS b
    ON a~matnr EQ b~matnr
    AND b~langu EQ sy-langu
    INTO CORRESPONDING FIELDS OF ps_row_0100
    WHERE a~warid EQ pt_info_0220-warid.

  IF sy-subrc <> 0.
    MESSAGE i999(zmcss) WITH '선택된 행의 자재가 없습니다.'.
    pv_subrc = 4.
    RETURN.
  ENDIF.

  LOOP AT pt_info_0220.
    CASE pt_info_0220-wtype_t.
      WHEN '불량'.
        MOVE-CORRESPONDING pt_info_0220 TO pt_info_0400.
        " WTYPE WTYPE_T MEINS WARID WDATE

        pt_info_0400-bdamt = pt_info_0220-amount.
        pt_info_0400-deamt = pt_info_0220-amount.
        pt_info_0400-adamt = '0'.

        pt_info_0400-matnr = ps_row_0100-matnr.
        APPEND pt_info_0400.
    ENDCASE.
  ENDLOOP.

  IF pt_info_0400[] IS INITIAL.
    MESSAGE i999(zmcss) WITH '선택된 불량 재고가 없습니다.'.
    pv_subrc = 4.
    RETURN.
  ENDIF.
  pv_subrc = 0.


ENDFORM.

FORM modify_screen_0100 .

  DATA ls_info_0100 LIKE gs_info_0100.
  DATA lv_no_fstck VALUE 'X'.

    ENDCASE.
  ENDLOOP.

  CASE lv_no_fstck.
    WHEN 'X'.
      LOOP AT SCREEN.
        CASE screen-group1.

        MODIFY SCREEN.
      ENDLOOP.
    WHEN OTHERS.
      CASE screen-group1.
        WHEN 'BTN'.
          screen-active = '1'.
      ENDCASE.
      MODIFY SCREEN.
  ENDCASE.

ENDFORM.
