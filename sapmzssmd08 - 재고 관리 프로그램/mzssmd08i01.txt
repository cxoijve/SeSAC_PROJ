MODULE exit INPUT.
  CASE ok_code.
    WHEN 'CANC'.
      LEAVE TO SCREEN 0.
    WHEN 'EXIT'.
      LEAVE PROGRAM.
    WHEN OTHERS.
  ENDCASE.
ENDMODULE.

MODULE user_command_0100 INPUT.
  CASE ok_code.
    WHEN 'BACK'.
      LEAVE TO SCREEN 0.
    WHEN 'SEARCH'.
      PERFORM search_0100.
    WHEN 'C200'.
      PERFORM c200_0100.
    WHEN 'DELALL'. " 불량 재고 일괄 폐기

    WHEN 'DELROW'. " 선택한 불량 재고 폐기
  ENDCASE.
ENDMODULE.

MODULE user_command_0200 INPUT.
