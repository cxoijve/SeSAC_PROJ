MODULE USER_COMMAND_0100 INPUT.

  CASE OK_CODE.
      GT_REV_REASON = VALUE #(
        ( CODE = '01' TEXT = '입력 오류' )
        ( CODE = '02' TEXT = '중복 입력' )
        ( CODE = '03' TEXT = '계정 오류' )
        ( CODE = '04' TEXT = '수정으로 인한 역분개' )
      ).


    WHEN 'BACK'.
      LEAVE TO SCREEN 0.

    WHEN 'ENTER'.
      " CODE에 따른 TEXT 를 찾아 ZSSSFI0301-RRSON_TEXT에 셋팅
      DATA: LS_REASON TYPE TY_REV_REASON.
      READ TABLE GT_REV_REASON INTO LS_REASON WITH KEY CODE = ZSSSFI0301-RRSON_T.
      IF SY-SUBRC = 0.
        ZSSSFI0301-RRSON_TEXT = LS_REASON-TEXT.
      ELSE.
        ZSSSFI0301-RRSON_TEXT = ''.
      ENDIF.

      ZSSSFI0301-FYEAR = ZSSSFI0301-JENUM+1(4).



    WHEN 'SEARCH'.
      CALL TRANSACTION 'ZSSFI03'.

    WHEN 'REV'.

      CALL SCREEN 110 STARTING AT 5 5 ENDING AT 30 20.

*      PERFORM CHECK_DATA_0110.

    WHEN 'CLEAR'.
      CLEAR : ZSSSFI0301-FYEAR.



    WHEN OTHERS.

  ENDCASE.
ENDMODULE.
