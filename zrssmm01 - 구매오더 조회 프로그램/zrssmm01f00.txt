*----------------------------------------------------------------------*
FORM get_plant_name  USING   VALUE(p_id) TYPE ztssplant-plant
                     CHANGING p_name TYPE ztssplant-pname.

  SELECT SINGLE pname
    INTO p_name
    FROM ztssplant
    WHERE plant = p_id.

ENDFORM.
