MODULE user_command_0100 INPUT.

  CASE ok_code.
    WHEN 'BACK'.
      LEAVE TO SCREEN 0.
    WHEN 'SEARCH'.
      PERFORM check_field_condition.
      PERFORM check_alv_condition.
      PERFORM get_customer_data.

    WHEN 'REGIST'.
      CALL SCREEN 200.
      CLEAR: zsssmd04003,
             zsssmd04004.

    WHEN 'CHANGE'.
      PERFORM get_detail_info CHANGING gv_subrc.
      IF gv_subrc IS INITIAL.
        CALL SCREEN 300.
        CLEAR: gs_data,
               zsssmd04006,
               zsssmd04007.
      ENDIF.

    WHEN 'DETAIL'.

    WHEN 'ENTER'.
      PERFORM check_field_condition.

*	WHEN OTHERS.
  ENDCASE.

ENDMODULE.

MODULE user_command_0200 INPUT.

  CASE ok_code.
    WHEN 'BACK'.
      LEAVE TO SCREEN 0.

    WHEN 'ENDSET'.

    WHEN 'INSERT'.
      PERFORM insert_email_data USING zsssmd04003-ednme.

      PERFORM insert_customer_data USING    zsssmd04003-endda
                                   CHANGING zsssmd04003-endda
                                            zsssmd04003.
  ENDCASE.

ENDMODULE.
