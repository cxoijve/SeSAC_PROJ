    FROM ztsssacat

  CLEAR: gt_cstid[],
         gt_goods[].

  IF zsssmd06002-cstid IS NOT INITIAL.
    gt_cstid-sign = 'I'.
    gt_cstid-option = 'EQ'.

  ENDIF.
