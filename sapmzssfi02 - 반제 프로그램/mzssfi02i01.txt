MODULE USER_COMMAND_0100 INPUT.

  CASE OK_CODE.

    WHEN 'BACK'.
      LEAVE TO SCREEN 0.

    WHEN 'SEARCH'.

      IF ZSBFI02-COND1 = 'X'.
        CALL SCREEN 0400.
      ELSE.
        CALL SCREEN 0500.
      ENDIF.
    WHEN 'DETAIL'.
      CALL SCREEN 0200.

    WHEN OTHERS.

  ENDCASE.

ENDMODULE.
