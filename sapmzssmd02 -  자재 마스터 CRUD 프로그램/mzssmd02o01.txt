*&---------------------------------------------------------------------*
*       text

MODULE create_alv_0100 OUTPUT.

  " alv 생성
  IF go_con_0100 IS INITIAL.
    PERFORM create_alv  USING     'CON'
                        CHANGING  go_con_0100
                                  go_alv_0100.
  ENDIF.

ENDMODULE.

*&      Module  DISPLAY_ALV_0100  OUTPUT
*&---------------------------------------------------------------------*
*       text
*----------------------------------------------------------------------*
MODULE display_alv_0100 OUTPUT.

  " alv display
  IF go_con_0100 IS NOT INITIAL.
    PERFORM display_alv  TABLES  gt_info_0100
                                 gt_fcat_0100
                                 gt_sort_0100
                         USING   'ZSSSMD0202'
                                 gs_layo_0100
                                 go_alv_0100.
  ENDIF.

ENDMODULE.
