MODULE EXIT INPUT.

  CASE OK_CODE.
    WHEN 'EXIT'.
      LEAVE PROGRAM.

    WHEN 'CANC'.
      LEAVE TO SCREEN 0.
  ENDCASE.

ENDMODULE.

MODULE USER_COMMAND_0100 INPUT.

  CASE OK_CODE.

    WHEN 'BACK'.
      LEAVE TO SCREEN 0.

    WHEN 'SAVE'.


    WHEN 'SEARCH'.
      CALL SCREEN 0200.

    WHEN 'LIST'.
      CALL SCREEN 0300.




    WHEN OTHERS.

  ENDCASE.

ENDMODULE.

MODULE USER_COMMAND_0200 INPUT.

  CASE OK_CODE.

    WHEN 'BACK'.
      LEAVE TO SCREEN 0.

    WHEN 'EXE'.
      PERFORM GET_NON_REV_DATA.

    WHEN 'REV'.
      CALL SCREEN 210 STARTING AT 5 5 ENDING AT 60 30.


    WHEN OTHERS.

  ENDCASE.

ENDMODULE.
