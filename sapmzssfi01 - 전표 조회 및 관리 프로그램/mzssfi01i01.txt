      PERFORM GET_RGTOR_NAME USING ZSSSFI0101-RGTOR

      " 헤더 정보 변경 버튼
      PERFORM CHANGE_ALV_MODE.

      PERFORM SET_DATA_0100.

      PERFORM GET_DATA.

      PERFORM SET_DATA_0200.

      CALL SCREEN 0200.

    WHEN 'DETAIL'.

      PERFORM SEL_ROWS.

      PERFORM SET_DATA_0300.

      CALL SCREEN 0300.

    WHEN 'INIT'.
      " 새로고침



    WHEN OTHERS.

  ENDCASE.

MODULE USER_COMMAND_0200 INPUT.
*  TABSTRIP-ACTIVETAB = OK_CODE.

  CASE OK_CODE.
    WHEN 'BACK'.
      LEAVE TO SCREEN 0.

    WHEN 'ENTER'.

      " 등록자명 취득
      PERFORM GET_RGTOR_NAME USING ZSSSFI0102-RGTOR
                             CHANGING ZSSSFI0102-RGTOR_T.

      " 등록자 조건: 회계부서 사원만 가능
      IF ZSSSFI0102-RGTOR IS NOT INITIAL.
        IF GS_EMP-DEPID <> 'D001'.
          MESSAGE S999(ZMCSS) WITH '회계부서 사원만 가능합니다. '
         DISPLAY LIKE 'E'.
        ENDIF.
      ENDIF.


      " 필수값 유효성 체크
      IF ZSSSFI0102-RGTOR IS INITIAL OR
         ZSSSFI0102-JTYPE IS INITIAL OR
         ZSSSFI0102-RGDAT IS INITIAL.
        MESSAGE S999(ZMCSS) WITH '등록자, 전표유형, 전기일을 모두 입력해 주세요.'
         DISPLAY LIKE 'E'.
      ENDIF.




    WHEN 'INSERT'.

      " 채번 후 생성
      PERFORM CHECK_DATA_0200 CHANGING GV_SUBRC.
      IF GV_SUBRC = 4.
        CLEAR GV_SUBRC.
        RETURN.
      ENDIF.

      PERFORM INSERT_DATA.


    WHEN 'CLEAR'.
      " 초기화

    WHEN 'BADD'.
      IF ZSSSFI0102-RGTOR IS INITIAL
        OR ZSSSFI0102-JTYPE IS INITIAL
        OR ZSSSFI0102-RGDAT IS INITIAL.


        MESSAGE S999(ZMCSS) WITH '등록자, 전표유형, 전기일을 모두 입력해 주세요.'
              DISPLAY LIKE 'E'.
        RETURN.
      ENDIF.



      " 라인아이템 추가
      CLEAR : ZSSSFI0103-JENUM,
              ZSSSFI0103-FYEAR,
              ZSSSFI0103-JESEQ,
              ZSSSFI0103-DEBIT,
              ZSSSFI0103-CREDIT.

      CALL SCREEN 210 STARTING AT 5 5 ENDING AT 120 30.


    WHEN 'BEDIT'.
      " 라인아이템 수정

    WHEN 'BDEL'.
      " 라인아이템 삭제

    WHEN OTHERS.
  ENDCASE.
ENDMODULE.
