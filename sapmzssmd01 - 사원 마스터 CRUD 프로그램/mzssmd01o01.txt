MODULE status_0300 OUTPUT.
  SET PF-STATUS 'S300'.
  SET TITLEBAR 'T300' WITH sy-uname.

*  CLEAR gv_first.
  " persg 라디오버튼 - 기존 정보로 설정
  IF gv_first IS INITIAL.
    CLEAR: zsssmd0105-persg_1, zsssmd0105-persg_2, zsssmd0105-persg_3.
    CASE gs_list-persg.
      WHEN '1'. zsssmd0105-persg_1 = 'X'.
      WHEN '2'. zsssmd0105-persg_2 = 'X'.
      WHEN '3'. zsssmd0105-persg_3 = 'X'.
    ENDCASE.

    " estat 라디오버튼 - 기존 정보로 설정
    CLEAR: zsssmd0105-estat_1, zsssmd0105-estat_2, zsssmd0105-estat_3.
    CASE gs_list-estat.
      WHEN '1'. zsssmd0105-estat_1 = 'X'.
      WHEN '2'. zsssmd0105-estat_2 = 'X'.
      WHEN '3'. zsssmd0105-estat_3 = 'X'.
    ENDCASE.
    " 플래그 세팅
    gv_first = abap_true.

    " telnr 나누기
    CLEAR: gv_tel_1, gv_tel_2, gv_tel_3.
    IF gs_list-telnr IS NOT INITIAL.
      SPLIT gs_list-telnr AT '-' INTO gv_tel_1 gv_tel_2 gv_tel_3.
    ENDIF.
    CLEAR: gv_tel_4, gv_tel_5, gv_tel_6.
    IF gs_list-telnr IS NOT INITIAL.
      SPLIT gs_list-telnr AT '-' INTO gv_tel_4 gv_tel_5 gv_tel_6.
    ENDIF.
  ENDIF.
ENDMODULE.
