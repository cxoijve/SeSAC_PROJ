MODULE user_command_0100 INPUT.

  CASE ok_code.
    WHEN 'BACK'.
      LEAVE TO SCREEN 0.
    WHEN 'TAB1' OR 'TAB2'.
      my_tab-activetab = ok_code.
    WHEN 'SEARCH'.
      IF my_tab-activetab = 'TAB1'.
        PERFORM get_tables_data.
        PERFORM check_tab1_field.
        PERFORM check_alv_condition.
        PERFORM get_iv_data.
        CALL SCREEN 200.
        CLEAR zsssmm03001.
      ENDIF.
    WHEN 'ENTER'.

  ENDCASE.

ENDMODULE.
*&---------------------------------------------------------------------*
*&      Module  EXIT  INPUT
*&---------------------------------------------------------------------*

*----------------------------------------------------------------------*
MODULE exit INPUT.

  CASE sy-ucomm.
    WHEN 'EXIT'.
      LEAVE PROGRAM.
    WHEN 'CANC'.
      LEAVE TO SCREEN 0.
  ENDCASE.

ENDMODULE.

MODULE user_command_0200 INPUT.

  CASE ok_code.
    WHEN 'BACK'.
      LEAVE TO SCREEN 0.
    WHEN 'REGIST'.
    WHEN 'CHANGE'.
  ENDCASE.

ENDMODULE.
