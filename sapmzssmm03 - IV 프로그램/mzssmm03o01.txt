MODULE status_0200 OUTPUT.

  ELSEIF my_tab-activetab = 'TAB2'.

  ELSEIF my_tab-activetab = 'TAB3'.
    APPEND 'REGIST' TO gt_excl.
    APPEND 'CHANGE' TO gt_excl.
    APPEND 'GRTAIL' TO gt_excl.
  ENDIF.

  SET PF-STATUS 'S200' EXCLUDING gt_excl.
  SET TITLEBAR 'T200'.
ENDMODULE.

    WHEN 'TAB1'.

    WHEN 'TAB3'.
