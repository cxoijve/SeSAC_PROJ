MODULE USER_COMMAND_0100 INPUT.

  CASE OK_CODE.
    WHEN 'BACK'.
      LEAVE TO SCREEN 0.

    WHEN 'ENTER'.

    WHEN 'SEARCH'.

      PERFORM GET_DATA.

    WHEN 'REGIST'.

      CALL SCREEN 0200.


    WHEN OTHERS.
  ENDCASE.
