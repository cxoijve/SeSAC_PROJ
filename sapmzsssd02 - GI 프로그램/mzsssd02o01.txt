MODULE set_ts_dynnr_0100 OUTPUT.
  CLEAR gv_dynnr.
  CASE ts_0100-activetab.
    WHEN 'T101'.
      gv_dynnr = '101'.
    WHEN 'T102'.
      gv_dynnr = '102'.
    WHEN 'T103'.
      gv_dynnr = '103'.
    WHEN 'T104'.
      gv_dynnr = '104'.

      " Default Tab
    WHEN OTHERS.
      gv_dynnr = '102'.
      ts_0100-activetab = 'T102'.
  ENDCASE.

ENDMODULE.
